import Player from '../entities/Player.js';
import SpawnSystem from '../systems/SpawnSystem.js';
import CollisionSystem from '../systems/CollisionSystem.js';
import EventBus from '../utils/EventBus.js';

export default class GameScene extends Phaser.Scene {
  constructor() {
    super({ key: 'GameScene' });
  }

  create() {
    // Background
    this.add.image(400, 250, 'background').setScrollFactor(0);

    // Player setup
    this.player = new Player(this, 200, 300);
    this.add.existing(this.player);

    // Systems setup
    this.spawnSystem = new SpawnSystem(this);
    this.collisionSystem = new CollisionSystem(this);

    // Event listeners
    EventBus.on('score:add', this.handleScoreAdd, this);
    EventBus.on('player:hit', this.handlePlayerHit, this);

    // Background music
    this.bgMusic = this.sound.add('backgroundMusic', { loop: true });
    this.bgMusic.play();

    // Camera setup to follow player
    this.cameras.main.startFollow(this.player);

    // Initialize score and lives
    this.score = 0;
    this.lives = 3;
  }

  update(time, delta) {
    this.player.update(time, delta);
    this.spawnSystem.update(time, delta);
    this.collisionSystem.update();
  }

  handleScoreAdd(points) {
    this.score += points;
  }

  handlePlayerHit() {
    this.lives -= 1;
    if (this.lives <= 0) {
      this.scene.start('GameOverScene', { score: this.score });
    }
  }
}

export default class Player extends Phaser.Physics.Arcade.Sprite {
  constructor(scene, x, y) {
    super(scene, x, y, 'player');
    this.scene = scene;

    // Physics setup
    this.scene.physics.world.enable(this);
    this.setCollideWorldBounds(true);

    // Player speeds
    this.speed = 200;
    this.jumpSpeed = -350;

    // Animation states
    this.isSliding = false;
  }

  update() {
    const cursors = this.scene.input.keyboard.createCursorKeys();

    // Horizontal movement
    if (cursors.left.isDown) {
      this.setVelocityX(-this.speed);
    } else if (cursors.right.isDown) {
      this.setVelocityX(this.speed);
    } else {
      this.setVelocityX(0);
    }

    // Jumping
    if (cursors.up.isDown && this.body.touching.down) {
      this.setVelocityY(this.jumpSpeed);
      this.scene.sound.play('jumpSound');
    }

    // Sliding (Shift key for drift)

export default class SpawnSystem {
  constructor(scene) {
    this.scene = scene;
    this.spawnRate = 1000; // 1 second
    this.lastSpawned = 0;
  }

  update(time) {
    if (time > this.lastSpawned + this.spawnRate) {
      this.spawnRandomObject();
      this.lastSpawned = time;
    }
  }

  spawnRandomObject() {
    const random = Phaser.Math.RND.integerInRange(1, 3);

    let object;
    if (random === 1) {
      object = this.scene.physics.add.sprite(800, 300, 'energyDrink').setScale(0.5);
    } else if (random === 2) {
      object = this.scene.physics.add.sprite(800, 300, 'curbObstacle').setScale(0.7);
    } else {
      object = this.scene.physics.add.sprite(800, 300, 'trashCanObstacle').setScale(0.5);
    }

    this.scene.physics.add.overlap(this.scene.player, object, () => {
      if (object.texture.key === 'energyDrink') {
        EventBus.emit('score:add', 10);
        this.scene.sound.play('drinkSound');
      } else { // Hit obstacle
        EventBus.emit('player:hit');
        this.scene.sound.play('crashSound');
      }
      object.destroy();
    });
  }
}

export default class SamuraiBoss extends Phaser.Physics.Arcade.Sprite {
  constructor(scene, x, y) {
    super(scene, x, y, 'samuraiBoss');
    this.scene = scene;

    // Physics setup
    this.scene.physics.world.enable(this);
    this.setCollideWorldBounds(true);

    // State machine
    this.states = {
      IDLE: 'idle',
      ATTACKING: 'attacking',
      DEAD: 'dead',
    };
    this.currentState = this.states.IDLE;

    // Settings
    this.health = 5;
  }

  update() {
    if (this.currentState === this.states.IDLE) {
      this.setVelocityX(-100);
    }

    if (this.currentState === this.states.ATTACKING) {
      this.setVelocityX(-200);
    }

    if (this.health <= 0) {
      this.currentState = this.states.DEAD;
      this.destroy();
      this.scene.sound.play('collisionSfx');
    }
  }

  takeDamage() {
    this.health -= 1;
    if (this.health <= 0) {
      this.currentState = this.states.DEAD;
    }
  }
}
